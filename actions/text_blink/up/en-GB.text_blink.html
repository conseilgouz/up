Blinking text

syntax 1 : {up text-blink}content{/up text-blink} // single flash
syntax 2 : {up text-blink=color:red}content{/up text-blink} // alternating style
syntax 3 : {up text-blink=prefset}content{/up text-blink} // set of options in prefs.ini
syntax 4 : {up text-blink=css_class}content{/up text-blink} // classe CSS
